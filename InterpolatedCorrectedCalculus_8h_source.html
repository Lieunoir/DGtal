83 DenseMatrix quadrangleSharpLocalMatrix( Face f ) const
84 {
85 DenseMatrix Basis = DenseMatrix::Zero( 3, 3 );
86 DenseMatrix Gavg = DenseMatrix::Zero( 3, 4 );
87 DenseMatrix Gedge = DenseMatrix::Zero( 3, 4 );
88 DenseMatrix S = DenseMatrix::Zero( 3, 4 );
89 const auto vtcs = myMesh->incidentVertices( f );
90 const auto i = vtcs[ 0 ];
91 const auto j = vtcs[ 1 ];
92 const auto k = vtcs[ 2 ];
93 const auto l = vtcs[ 3 ];
94 const auto dxij = myMesh->position( j ) - myMesh->position( i );
95 const auto dxjk = myMesh->position( k ) - myMesh->position( j );
96 const auto xij = dxij.getNormalized();
97 const auto xjk = dxjk.getNormalized();
98
99 const RealVector n = xij.crossProduct( xjk );
100 const Dimension x = fabs( xij[ 0 ] ) > fabs( xij[ 1 ] )
101 ? ( fabs( xij[ 0 ] ) > fabs( xij[ 2 ] ) ? 0 : 2 )
102 : ( fabs( xij[ 1 ] ) > fabs( xij[ 2 ] ) ? 1 : 2 );
103 const Dimension y = fabs( xjk[ 0 ] ) > fabs( xjk[ 1 ] )
104 ? ( fabs( xjk[ 0 ] ) > fabs( xjk[ 2 ] ) ? 0 : 2 )
105 : ( fabs( xjk[ 1 ] ) > fabs( xjk[ 2 ] ) ? 1 : 2 );
106 const Dimension z = fabs( n[ 0 ] ) > fabs( n[ 1 ] )
107 ? ( fabs( n[ 0 ] ) > fabs( n[ 2 ] ) ? 0 : 2 )
108 : ( fabs( n[ 1 ] ) > fabs( n[ 2 ] ) ? 1 : 2 );
109 if ( ( x == y ) || ( x == z ) || ( y == z ) || ( x + y + z > 3 ) )
110 trace.warning() << "Bad basis: " << x << y << z << std::endl;
111 const Scalar sign_x = xij[ x ] > 0.0 ? 1.0 : -1.0;
112 const Scalar sign_y = xjk[ y ] > 0.0 ? 1.0 : -1.0;
113 const Scalar sign_z = n[ z ] > 0.0 ? 1.0 : -1.0;
114
115 if ( use2ndOrder == false )
116 {
117 const auto ui = myMesh->vertexNormal( i );
118 const auto uj = myMesh->vertexNormal( j );
119 const auto uk = myMesh->vertexNormal( k );
120 const auto ul = myMesh->vertexNormal( l );
121 const auto u0 = 0.5 * ( ui + uj );
122 const auto u1 = 0.5 * ( uj + uk );
123 const auto u2 = 0.5 * ( uk + ul );
124 const auto u3 = 0.5 * ( ul + ui );
125 const auto u = 0.5 * ( u0 + u2 );
126 // We put u into the "natural" frame of the surfel
127 const auto ux = xij.dot( u );
128 const auto uy = xjk.dot( u );
129 const auto uz = n.dot( u );
130 Gavg.coeffRef( 0, 0 ) = uz;
131 Gavg.coeffRef( 0, 2 ) = -uz;
132 Gavg.coeffRef( 1, 1 ) = uz;
133 Gavg.coeffRef( 1, 3 ) = -uz;
134 Gavg.coeffRef( 2, 0 ) = -ux;
135 Gavg.coeffRef( 2, 1 ) = -uy;
136 Gavg.coeffRef( 2, 2 ) = ux;
137 Gavg.coeffRef( 2, 3 ) = uy;
138 Gavg /= 3.0;
139 Gedge.coeffRef( 0, 0 ) = n.dot( u0 ); // u0[z];
140 Gedge.coeffRef( 0, 2 ) = -n.dot( u2 ); // -u2[z];
141 Gedge.coeffRef( 1, 1 ) = n.dot( u1 ); // u1[z];
142 Gedge.coeffRef( 1, 3 ) = -n.dot( u3 ); // -u3[z];
143 Gedge.coeffRef( 2, 0 ) = -xij.dot( u0 ); // -u0[x];
144 Gedge.coeffRef( 2, 1 ) = -xjk.dot( u1 ); // -u1[y];
145 Gedge.coeffRef( 2, 2 ) = xij.dot( u2 ); // u2[x];
146 Gedge.coeffRef( 2, 3 ) = xjk.dot( u3 ); // u3[y];
147 Gedge /= 6.0;
148 // We compute the change of basis to come back to the canonic frame of
149 // R3.
150 S = Gavg + Gedge;
151 }
152 else
153 { // use2ndOrder
154 const auto u00 = myMesh->vertexNormal( i );
155 const auto u10 = myMesh->vertexNormal( j );
156 const auto u11 = myMesh->vertexNormal( k );
157 const auto u01 = myMesh->vertexNormal( l );
158 const auto ux00 = xij.dot( u00 );
159 const auto ux10 = xij.dot( u10 );
160 const auto ux11 = xij.dot( u11 );
161 const auto ux01 = xij.dot( u01 );
162 const auto uy00 = xjk.dot( u00 );
163 const auto uy10 = xjk.dot( u10 );
164 const auto uy11 = xjk.dot( u11 );
165 const auto uy01 = xjk.dot( u01 );
166 const auto uz00 = n.dot( u00 );
167 const auto uz10 = n.dot( u10 );
168 const auto uz11 = n.dot( u11 );
169 const auto uz01 = n.dot( u01 );
170 const auto a = 1.0 / ( u00 + u10 ).norm();
171 const auto b = 1.0 / ( u10 + u11 ).norm();
172 const auto c = 1.0 / ( u11 + u01 ).norm();
173 const auto d = 1.0 / ( u01 + u00 ).norm();
174 const auto e = 1.0 / ( u00 + u10 + u11 + u01 ).norm();
175 S.coeffRef( 0, 0 ) =
176 ( 4 * a + 2 * d + 8 * e + 1 ) * uz00 + 2 * ( d + 4 * e ) * uz01 +
177 ( 4 * a + 2 * b + 8 * e + 1 ) * uz10 + 2 * ( b + 4 * e ) * uz11;
178 S.coeffRef( 0, 2 ) =
179 -2 * ( d + 4 * e ) * uz00 - ( 4 * c + 2 * d + 8 * e + 1 ) * uz01 -
180 2 * ( b + 4 * e ) * uz10 - ( 2 * b + 4 * c + 8 * e + 1 ) * uz11;
181 S.coeffRef( 1, 1 ) = 2 * ( a + 4 * e ) * uz00 +
182 2 * ( c + 4 * e ) * uz01 +
183 ( 2 * a + 4 * b + 8 * e + 1 ) * uz10 +
184 ( 4 * b + 2 * c + 8 * e + 1 ) * uz11;
185 S.coeffRef( 1, 3 ) = -( 2 * a + 4 * d + 8 * e + 1 ) * uz00 -
186 ( 2 * c + 4 * d + 8 * e + 1 ) * uz01 -
187 2 * ( a + 4 * e ) * uz10 -
188 2 * ( c + 4 * e ) * uz11;
189 S.coeffRef( 2, 0 ) =
190 -( 4 * a + 2 * d + 8 * e + 1 ) * ux00 - 2 * ( d + 4 * e ) * ux01 -
191 ( 4 * a + 2 * b + 8 * e + 1 ) * ux10 - 2 * ( b + 4 * e ) * ux11;
192 S.coeffRef( 2, 1 ) = -2 * ( a + 4 * e ) * uy00 -
193 2 * ( c + 4 * e ) * uy01 -
194 ( 2 * a + 4 * b + 8 * e + 1 ) * uy10 -
195 ( 4 * b + 2 * c + 8 * e + 1 ) * uy11;
196 S.coeffRef( 2, 2 ) =
197 2 * ( d + 4 * e ) * ux00 + ( 4 * c + 2 * d + 8 * e + 1 ) * ux01 +
198 2 * ( b + 4 * e ) * ux10 + ( 2 * b + 4 * c + 8 * e + 1 ) * ux11;
199 S.coeffRef( 2, 3 ) = ( 2 * a + 4 * d + 8 * e + 1 ) * uy00 +
200 ( 2 * c + 4 * d + 8 * e + 1 ) * uy01 +
201 2 * ( a + 4 * e ) * uy10 +
202 2 * ( c + 4 * e ) * uy11;
203 S /= 36.0;
204 }
205 const Scalar h = fabs( dxij[ x ] );
206 Basis.coeffRef( 0, x ) = sign_x * h;
207 Basis.coeffRef( 1, y ) = sign_y * h;
208 Basis.coeffRef( 2, z ) = sign_z * h;
209 const auto a_f = quadrangleArea( f );
210 return ( Basis.transpose() / a_f ) * S;
211 }

213 DenseMatrix quadrangleFlatLocalMatrix( Face f ) const
214 {
215 DenseMatrix Basis = DenseMatrix::Zero( 3, 3 );
216 DenseMatrix V = DenseMatrix::Zero( 4, 3 );
217 const auto vtcs = myMesh->incidentVertices( f );
218 if ( vtcs.size() != 4 )
219 {
220 trace.error()
221 << "[SurfaceMeshDEC::quadrangleCorrectedDigitalFlatLocalMatrix]"
222 << " restricted to quadrangle faces." << std::endl;
223 return V;
224 }
225 const auto i = vtcs[ 0 ];
226 const auto j = vtcs[ 1 ];
227 const auto k = vtcs[ 2 ];
228 const auto l = vtcs[ 3 ];
229 const auto dxij = myMesh->position( j ) - myMesh->position( i );
230 const auto dxjk = myMesh->position( k ) - myMesh->position( j );
231 const auto xij = dxij.getNormalized();
232 const auto xjk = dxjk.getNormalized();
233
234 const RealVector n = xij.crossProduct( xjk );
235 const Dimension x = fabs( xij[ 0 ] ) > fabs( xij[ 1 ] )
236 ? ( fabs( xij[ 0 ] ) > fabs( xij[ 2 ] ) ? 0 : 2 )
237 : ( fabs( xij[ 1 ] ) > fabs( xij[ 2 ] ) ? 1 : 2 );
238 const Dimension y = fabs( xjk[ 0 ] ) > fabs( xjk[ 1 ] )
239 ? ( fabs( xjk[ 0 ] ) > fabs( xjk[ 2 ] ) ? 0 : 2 )
240 : ( fabs( xjk[ 1 ] ) > fabs( xjk[ 2 ] ) ? 1 : 2 );
241 const Dimension z = fabs( n[ 0 ] ) > fabs( n[ 1 ] )
242 ? ( fabs( n[ 0 ] ) > fabs( n[ 2 ] ) ? 0 : 2 )
243 : ( fabs( n[ 1 ] ) > fabs( n[ 2 ] ) ? 1 : 2 );
244 if ( ( x == y ) || ( x == z ) || ( y == z ) || ( x + y + z > 3 ) )
245 trace.warning() << "Bad basis: " << x << y << z << std::endl;
246 const Scalar sign_x = xij[ x ] > 0.0 ? 1.0 : -1.0;
247 const Scalar sign_y = xjk[ y ] > 0.0 ? 1.0 : -1.0;
248 const Scalar sign_z = n[ z ] > 0.0 ? 1.0 : -1.0;
249 const auto ui = myMesh->vertexNormal( i );
250 const auto uj = myMesh->vertexNormal( j );
251 const auto uk = myMesh->vertexNormal( k );
252 const auto ul = myMesh->vertexNormal( l );
253 // We put all the u into the "natural" frame of the surfel
254 const auto uix = xij.dot( ui );
255 const auto uiy = xjk.dot( ui );
256 const auto uiz = n.dot( ui );
257 const auto ujx = xij.dot( uj );
258 const auto ujy = xjk.dot( uj );
259 const auto ujz = n.dot( uj );
260 const auto ukx = xij.dot( uk );
261 const auto uky = xjk.dot( uk );
262 const auto ukz = n.dot( uk );
263 const auto ulx = xij.dot( ul );
264 const auto uly = xjk.dot( ul );
265 const auto ulz = n.dot( ul );
266 if ( !use2ndOrder )
267 {
268 V.coeffRef( 0, 0 ) =
269 6.0 - 2.0 * ( 2 * uix + ujx ) * uix - 2.0 * ( uix + 2 * ujx ) * ujx;
270 V.coeffRef( 0, 1 ) =
271 -2.0 * ( 2 * uix + ujx ) * uiy - 2.0 * ( uix + 2 * ujx ) * ujy;
272 V.coeffRef( 0, 2 ) =
273 -2.0 * ( 2 * uix + ujx ) * uiz - 2.0 * ( uix + 2 * ujx ) * ujz;
274 V.coeffRef( 1, 0 ) =
275 -2.0 * ( 2 * ujy + uky ) * ujx - 2.0 * ( ujy + 2 * uky ) * ukx;
276 V.coeffRef( 1, 1 ) =
277 6.0 - 2.0 * ( 2 * ujy + uky ) * ujy - 2.0 * ( ujy + 2 * uky ) * uky;
278 V.coeffRef( 1, 2 ) =
279 -2.0 * ( 2 * ujy + uky ) * ujz - 2.0 * ( ujy + 2 * uky ) * ukz;
280 V.coeffRef( 2, 0 ) =
281 -6.0 + 2.0 * ( 2 * ukx + ulx ) * ukx + 2.0 * ( ukx + 2 * ulx ) * ulx;
282 V.coeffRef( 2, 1 ) =
283 2.0 * ( 2 * ukx + ulx ) * uky + 2.0 * ( ukx + 2 * ulx ) * uly;
284 V.coeffRef( 2, 2 ) =
285 2.0 * ( 2 * ukx + ulx ) * ukz + 2.0 * ( ukx + 2 * ulx ) * ulz;
286 V.coeffRef( 3, 0 ) =
287 2.0 * ( 2 * uly + uiy ) * ulx + 2.0 * ( uly + 2 * uiy ) * uix;
288 V.coeffRef( 3, 1 ) =
289 -6.0 + 2.0 * ( 2 * uly + uiy ) * uly + 2.0 * ( uly + 2 * uiy ) * uiy;
290 V.coeffRef( 3, 2 ) =
291 2.0 * ( 2 * uly + uiy ) * ulz + 2.0 * ( uly + 2 * uiy ) * uiz;
292 V /= 6.0;
293 }
294 else
295 {
296 const auto a = 1.0 / ( ui + uj ).norm();
297 const auto b = 1.0 / ( uj + uk ).norm();
298 const auto c = 1.0 / ( uk + ul ).norm();
299 const auto d = 1.0 / ( ul + ui ).norm();
300 const auto e = 1.0 / ( ui + uj + uk + ul ).norm();
301 const auto ap = 4 * a * a + a + 1;
302 const auto app = 16 * a * a + 4 * a - 1;
303 const auto bp = 4 * b * b + b + 1;
304 const auto bpp = 16 * b * b + 4 * b - 1;
305 const auto cp = 4 * c * c + c + 1;
306 const auto cpp = 16 * c * c + 4 * c - 1;
307 const auto dp = 4 * d * d + d + 1;
308 const auto dpp = 16 * d * d + 4 * d - 1;
309 V.coeffRef( 0, 0 ) =
310 -4 * ap * uix * uix - 2 * app * uix * ujx - 4 * ap * ujx * ujx + 30.0;
311 V.coeffRef( 0, 1 ) = -( 4 * ap * uix + app * ujx ) * uiy -
312 ( app * uix + 4 * ap * ujx ) * ujy;
313 V.coeffRef( 0, 2 ) = -( 4 * ap * uix + app * ujx ) * uiz -
314 ( app * uix + 4 * ap * ujx ) * ujz;
315 V.coeffRef( 1, 0 ) = -( 4 * bp * ujx + bpp * ukx ) * ujy -
316 ( bpp * ujx + 4 * bp * ukx ) * uky;
317 V.coeffRef( 1, 1 ) =
318 -4 * bp * ujy * ujy - 2 * bpp * ujy * uky - 4 * bp * uky * uky + 30.0;
319 V.coeffRef( 1, 2 ) = -( 4 * bp * ujy + bpp * uky ) * ujz -
320 ( bpp * ujy + 4 * bp * uky ) * ukz;
321 V.coeffRef( 2, 0 ) =
322 4 * cp * ulx * ulx + 2 * cpp * ulx * ukx + 4 * cp * ukx * ukx - 30.0;
323 V.coeffRef( 2, 1 ) =
324 ( 4 * cp * ulx + cpp * ukx ) * uly + ( cpp * ulx + 4 * cp * ukx ) * uky;
325 V.coeffRef( 2, 2 ) =
326 ( 4 * cp * ulx + cpp * ukx ) * ulz + ( cpp * ulx + 4 * cp * ukx ) * ukz;
327 V.coeffRef( 3, 0 ) =
328 ( 4 * dp * uix + dpp * ulx ) * uiy + ( dpp * uix + 4 * dp * ulx ) * uly;
329 V.coeffRef( 3, 1 ) =
330 4 * dp * uiy * uiy + 2 * dpp * uiy * uly + 4 * dp * uly * uly - 30.0;
331 V.coeffRef( 3, 2 ) =
332 ( 4 * dp * uiy + dpp * uly ) * uiz + ( dpp * uiy + 4 * dp * uly ) * ulz;
333 V /= 30.0;
334 }
335
336 // We compute the change of basis to come back to the canonic frame of R3.
337 const Scalar h = fabs( dxij[ x ] );
338 Basis.coeffRef( 0, x ) = sign_x * h;
339 Basis.coeffRef( 1, y ) = sign_y * h;
340 Basis.coeffRef( 2, z ) = sign_z * h;
341 return V * Basis;
342 }

345 DenseMatrix quadrangleInnerProduct0( Face f ) const
346 {
347 const auto vtcs = myMesh->incidentVertices( f );
348 const auto i = vtcs[ 0 ];
349 const auto j = vtcs[ 1 ];
350 const auto k = vtcs[ 2 ];
351 const auto l = vtcs[ 3 ];
352 const auto xij = myMesh->position( j ) - myMesh->position( i );
353 const auto xil = myMesh->position( l ) - myMesh->position( i );
354 const auto xlk = myMesh->position( k ) - myMesh->position( l );
355 const auto xjk = myMesh->position( k ) - myMesh->position( j );
356 const RealVector nn[ 4 ] = { /* ni */ xij.crossProduct( xil ),
357 /* nj */ xij.crossProduct( xjk ),
358 /* nk */ xlk.crossProduct( xjk ),
359 /* nl */ xlk.crossProduct( xil ) };
360 const Scalar factor = 1.0 / 3600.0;
361 const Vertex id[ 4 ] = { i, j, k, l };
362 DenseMatrix M = DenseMatrix::Zero( 4, 4 );
363 if ( !use2ndOrder )
364 {
365 // All coefs are computed from polynomial of the form:
366 // s^a (1-s)^a t^b (1-t)^b
367 // Indices are 0,1,2,3 == i,j,k,l
368 const int pa[ 4 ] = { 0, 1, 1, 0 };
369 const int pb[ 4 ] = { 0, 0, 1, 1 };
370 const double co[ 5 ][ 5 ] = { { 144.0, 36.0, 24.0, 36.0, 144.0 },
371 { 36.0, 9.0, 6.0, 9.0, 36.0 },
372 { 24.0, 6.0, 4.0, 6.0, 24.0 },
373 { 36.0, 9.0, 6.0, 9.0, 36.0 },
374 { 144.0, 36.0, 24.0, 36.0, 144.0 } };
375 for ( int g = 0; g < 4; ++g ) // left 0-form at vertex
376 for ( int h = 0; h < 4; ++h ) // right 0-form at vertex
377 { // Computing coefficients for ( id[ g ], id[ h ] )
378 for ( int u = 0; u < 4; ++u ) // corrected normal at vertex
379 for ( int n = 0; n < 4; ++n ) // naive normal at vertex
380 {
381 const int a = pa[ g ] + pa[ h ] + pa[ u ] + pa[ n ];
382 const int b = pb[ g ] + pb[ h ] + pb[ u ] + pb[ n ];
383 const double coeff =
384 factor * co[ a ][ b ] *
385 myMesh->vertexNormal( id[ u ] ).dot( nn[ n ] );
386 M( g, h ) += coeff;
387 }
388 }
389 }
390 else
391 { // use2ndOrder
392 // # 1ere ligne de M0 (avec kle facteur 3600*)
393 const auto ui = myMesh->vertexNormal( i );
394 const auto uj = myMesh->vertexNormal( j );
395 const auto uk = myMesh->vertexNormal( k );
396 const auto ul = myMesh->vertexNormal( l );
397 // formula is theoretically valid for constant geometric normals nn
398 const auto uiz = nn[ 0 ].dot( myMesh->vertexNormal( i ) );
399 const auto ujz = nn[ 1 ].dot( myMesh->vertexNormal( j ) );
400 const auto ukz = nn[ 2 ].dot( myMesh->vertexNormal( k ) );
401 const auto ulz = nn[ 3 ].dot( myMesh->vertexNormal( l ) );
402 const auto a = 1.0 / ( ui + uj ).norm();
403 const auto b = 1.0 / ( uj + uk ).norm();
404 const auto c = 1.0 / ( uk + ul ).norm();
405 const auto d = 1.0 / ( ul + ui ).norm();
406 const auto e = 1.0 / ( ui + uj + uk + ul ).norm();
407 M.coeffRef( 0, 0 ) = ( 108 * a + 108 * d + 144 * e + 81 ) * uiz +
408 ( 108 * a - 12 * b + 144 * e - 9 ) * ujz +
409 ( -12 * b - 12 * c + 144 * e + 1 ) * ukz +
410 ( -12 * c + 108 * d + 144 * e - 9 ) * ulz;
411 M.coeffRef( 0, 1 ) = ( 72 * a + 12 * d + 96 * e + 9 ) * uiz +
412 ( 72 * a + 12 * b + 96 * e + 9 ) * ujz +
413 ( 12 * b - 8 * c + 96 * e - 1 ) * ukz +
414 ( -8 * c + 12 * d + 96 * e - 1 ) * ulz;
415 M.coeffRef( 0, 2 ) = ( 8 * a + 8 * d + 64 * e + 1 ) * uiz +
416 ( 8 * a + 8 * b + 64 * e + 1 ) * ujz +
417 ( 8 * b + 8 * c + 64 * e + 1 ) * ukz +
418 ( 8 * c + 8 * d + 64 * e + 1 ) * ulz;
419 M.coeffRef( 0, 3 ) = ( 12 * a + 72 * d + 96 * e + 9 ) * uiz +
420 ( 12 * a - 8 * b + 96 * e - 1 ) * ujz +
421 ( -8 * b + 12 * c + 96 * e - 1 ) * ukz +
422 ( 12 * c + 72 * d + 96 * e + 9 ) * ulz;
423 M.coeffRef( 1, 0 ) = M( 0, 1 );
424 M.coeffRef( 1, 1 ) = ( 108 * a - 12 * d + 144 * e - 9 ) * uiz +
425 ( 108 * a + 108 * b + 144 * e + 81 ) * ujz +
426 ( 108 * b - 12 * c + 144 * e - 9 ) * ukz +
427 ( -12 * c - 12 * d + 144 * e + 1 ) * ulz;
428 M.coeffRef( 1, 2 ) = ( 12 * a - 8 * d + 96 * e - 1 ) * uiz +
429 ( 12 * a + 72 * b + 96 * e + 9 ) * ujz +
430 ( 72 * b + 12 * c + 96 * e + 9 ) * ukz +
431 ( 12 * c - 8 * d + 96 * e - 1 ) * ulz;
432 M.coeffRef( 1, 3 ) = ( 8 * a + 8 * d + 64 * e + 1 ) * uiz +
433 ( 8 * a + 8 * b + 64 * e + 1 ) * ujz +
434 ( 8 * b + 8 * c + 64 * e + 1 ) * ukz +
435 ( 8 * c + 8 * d + 64 * e + 1 ) * ulz;
436 M.coeffRef( 2, 0 ) = M( 0, 2 );
437 M.coeffRef( 2, 1 ) = M( 1, 2 );
438 M.coeffRef( 2, 2 ) = ( -12 * a - 12 * d + 144 * e + 1 ) * uiz +
439 ( -12 * a + 108 * b + 144 * e - 9 ) * ujz +
440 ( 108 * b + 108 * c + 144 * e + 81 ) * ukz +
441 ( 108 * c - 12 * d + 144 * e - 9 ) * ulz;
442 M.coeffRef( 2, 3 ) = ( -8 * a + 12 * d + 96 * e - 1 ) * uiz +
443 ( -8 * a + 12 * b + 96 * e - 1 ) * ujz +
444 ( 12 * b + 72 * c + 96 * e + 9 ) * ukz +
445 ( 72 * c + 12 * d + 96 * e + 9 ) * ulz;
446 M.coeffRef( 3, 0 ) = M( 0, 3 );
447 M.coeffRef( 3, 1 ) = M( 1, 3 );
448 M.coeffRef( 3, 2 ) = M( 2, 3 );
449 M.coeffRef( 3, 3 ) = ( -12 * a + 108 * d + 144 * e - 9 ) * uiz +
450 ( -12 * a - 12 * b + 144 * e + 1 ) * ujz +
451 ( -12 * b + 108 * c + 144 * e - 9 ) * ukz +
452 ( 108 * c + 108 * d + 144 * e + 81 ) * ulz;
453 M *= factor;
454 }
455 return M;
456 }

458 Scalar quadrangleArea( Face f ) const
459 {
460 const auto vtcs = myMesh->incidentVertices( f );
461 ASSERT( vtcs.size() == 4 );
462 const auto i = vtcs[ 0 ];
463 const auto j = vtcs[ 1 ];
464 const auto k = vtcs[ 2 ];
465 const auto l = vtcs[ 3 ];
466 const auto xij = myMesh->position( j ) - myMesh->position( i );
467 const auto xil = myMesh->position( l ) - myMesh->position( i );
468 const auto xlk = myMesh->position( k ) - myMesh->position( l );
469 const auto xjk = myMesh->position( k ) - myMesh->position( j );
470 const RealVector ni = xij.crossProduct( xil );
471 const RealVector nj = xij.crossProduct( xjk );
472 const RealVector nk = xlk.crossProduct( xjk );
473 const RealVector nl = xlk.crossProduct( xil );
474 const auto & ui = myMesh->vertexNormal( i );
475 const auto & uj = myMesh->vertexNormal( j );
476 const auto & uk = myMesh->vertexNormal( k );
477 const auto & ul = myMesh->vertexNormal( l );
478 const Scalar factor = 1.0 / 36.0;
479 if ( !use2ndOrder )
480 return factor * ( ni.dot( 4 * ui + 2 * uj + uk + 2 * ul ) +
481 nj.dot( 2 * ui + 4 * uj + 2 * uk + ul ) +
482 nk.dot( ui + 2 * uj + 4 * uk + 2 * ul ) +
483 nl.dot( 2 * ui + uj + 2 * uk + 4 * ul ) );
484 else
485 {
486 // Works only for surfels ! i.e. xij==xlk and xjk==xil
487 // If we assume f00=f10=f01=f11=1, the corrected area of a surfel is,
488 // after *36 factor A = (4*a+4*d+16*e+1) < u00 | n >
489 // + (4*a+4*b+16*e+1) < u10 | n >
490 // + (4*b+4*c+16*e+1) < u11 | n >
491 // + (4*c+4*d+16*e+1) < u01 | n >
492 const auto a = 1.0 / ( ui + uj ).norm();
493 const auto b = 1.0 / ( uj + uk ).norm();
494 const auto c = 1.0 / ( uk + ul ).norm();
495 const auto d = 1.0 / ( ul + ui ).norm();
496 const auto e = 1.0 / ( ui + uj + uk + ul ).norm();
497 return factor * ( ( 4 * a + 4 * d + 16 * e + 1 ) * ni.dot( ui ) +
498 ( 4 * a + 4 * b + 16 * e + 1 ) * nj.dot( uj ) +
499 ( 4 * b + 4 * c + 16 * e + 1 ) * nk.dot( uk ) +
500 ( 4 * c + 4 * d + 16 * e + 1 ) * nl.dot( ul ) );
501 }
502 }

575 DenseMatrix buildLocalM1( Index f ) const
576 {
577 const auto a_f = quadrangleArea( f );
578 DenseMatrix CA = DenseMatrix::Identity( 3, 3 ) * a_f;
579 DenseMatrix sharp = buildLocalSharp( f );
580 DenseMatrix Id = DenseMatrix::Identity( 4, 4 );
581 DenseMatrix P = Id - buildLocalFlat( f ) * sharp;
582 // Compute inner product for 1-forms
583 DenseMatrix myM1 =
584 sharp.transpose() * CA * sharp + lambda * P.transpose() * P;
585 return myM1;
586 }

588 LinearOperator buildM1() const
589 {
590
591 LinearOperator CA( 3 * myMesh->nbFaces(), 3 * myMesh->nbFaces() );
592 Triplets triplets;
593 for ( Face f = 0; f < myMesh->nbFaces(); ++f )
594 {
595 const auto a_f = quadrangleArea( f );
596 if ( a_f <= 1e-8 )
597 trace.warning() << "Bad corrected area " << f << " " << a_f
598 << std::endl;
599 triplets.push_back( { 3 * f, 3 * f, a_f } );
600 triplets.push_back( { 3 * f + 1, 3 * f + 1, a_f } );
601 triplets.push_back( { 3 * f + 2, 3 * f + 2, a_f } );
602 }
603 CA.setFromTriplets( triplets.cbegin(), triplets.cend() );
604 // Compute projection operator
605 LinearOperator mySharp = buildSharp();
606 LinearOperator Id1( myMesh->nbEdges(), myMesh->nbEdges() );
607 Id1.setIdentity();
608 LinearOperator P = Id1 - buildFlat() * mySharp;
609 // Compute inner product for 1-forms
610 LinearOperator myM1 =
611 mySharp.transpose() * CA * mySharp + lambda * P.transpose() * P;
612 return myM1;
613 }

666 LinearOperator buildSharp() const
667 {
668 Triplets triplets;
669 LinearOperator mySharp =
670 LinearOperator( 3 * myMesh->nbFaces(), myMesh->nbEdges() );
671 for ( Face f = 0; f < myMesh->nbFaces(); ++f )
672 {
673 // std::cout << "Face " << f << std::endl;
674 const auto vtcs = myMesh->incidentVertices( f );
675 const auto nc = vtcs.size();
676 if ( nc != 4 )
677 {
678 trace.error() << "[InterpolatedCorrectedCalculus::buildSharp]"
679 << " restricted to quadrangle faces." << std::endl;
680 return mySharp;
681 }
682 Edge edge_idcs[ 4 ];
683 Scalar edge_sign[ 4 ];
684 for ( Size i = 0; i < nc; i++ )
685 {
686 edge_idcs[ i ] =
687 myMesh->makeEdge( vtcs[ i ], vtcs[ ( i + 1 ) % nc ] );
688 edge_sign[ i ] = vtcs[ i ] < vtcs[ ( i + 1 ) % nc ] ? 1.0 : -1.0;
689 }
690 const auto G_f = buildLocalSharp( f );
691 for ( Dimension x = 0; x < 3; ++x )
692 for ( Dimension n = 0; n < nc; ++n )
693 triplets.push_back(
694 { 3 * f + x, edge_idcs[ n ], edge_sign[ n ] * G_f( x, n ) } );
695 }
696 mySharp.setFromTriplets( triplets.cbegin(), triplets.cend() );
697 return mySharp;
698 }

739 triplets.push_back( { edge_idcs[ n ], 3 * f + x,
