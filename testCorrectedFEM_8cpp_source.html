83TEST_CASE( "Testing PolygonalCalculus and DirichletConditions" )
84{
85 typedef Shortcuts< KSpace > SH3;
86 typedef SurfaceMesh< RealPoint,RealPoint > Mesh;
87 typedef Mesh::Index Index;
88 typedef NormalCorrectedFEM<EigenLinearAlgebraBackend, RealPoint,RealVector > CFEM;
89 typedef CFEM::LinearAlgebraBackend::DenseVector DenseVector;
90 typedef DirichletConditions< EigenLinearAlgebraBackend > DC;
91
92 // Build a more complex surface.
93 auto params = SH3::defaultParameters();
94
95 params( "polynomial", "0.1*y*y -0.1*x*x - 2.0*z" )( "gridstep", 2.0 );
96 auto implicit_shape = SH3::makeImplicitShape3D ( params );
97 auto digitized_shape = SH3::makeDigitizedImplicitShape3D( implicit_shape, params );
98 auto K = SH3::getKSpace( params );
99 auto binary_image = SH3::makeBinaryImage( digitized_shape, params );
100 auto surface = SH3::makeDigitalSurface( binary_image, K, params );
101 auto primalSurface = SH3::makePrimalSurfaceMesh(surface);
102
103 std::vector<std::vector< Index > > faces;
104 std::vector<RealPoint> positions = primalSurface->positions();
105 for( Index face= 0 ; face < primalSurface->nbFaces(); ++face)
106 faces.push_back(primalSurface->incidentVertices( face ));
107
108 Mesh surfmesh = Mesh( positions.begin(), positions.end(),
109 faces.begin(), faces.end() );
110 auto boundaryEdges = surfmesh.computeManifoldBoundaryEdges();
111 surfmesh.computeFaceNormalsFromPositions();
112
113 SECTION("Check surface")
114 {
115 REQUIRE( surfmesh.nbVertices() == 1364 );
116 REQUIRE( surfmesh.nbFaces() == 1279 );
117 REQUIRE( boundaryEdges.size() == 168 );
118 }
119
120 // Builds calculus and solve a Poisson problem with Dirichlet boundary conditions
121 CFEM calculus( surfmesh );
122 // Laplace opeartor
123 CFEM::LinearOperator L = calculus.L0();
124 // value on boundary
125 CFEM::LinearAlgebraBackend::DenseVector g = CFEM::LinearAlgebraBackend::DenseVector::Zero(surfmesh.nbVertices());
126 // characteristic set of boundary
127 DC::IntegerVector b = DC::IntegerVector::Zero( g.rows() );
128
129 SECTION("Solve Poisson problem with boundary Dirichlet conditions")
130 {
131 for ( double scale = 0.1; scale < 2.0; scale *= 2.0 )
132 {
133 std::cout << "scale=" << scale << std::endl;
134 auto phi = [&]( Index v)
135 {
136 return cos(scale*(surfmesh.position(v)[0]))
137 * (scale*surfmesh.position(v)[1]);
138 };
139
140 for(auto &e: boundaryEdges)
141 {
142 auto adjVertices = surfmesh.edgeVertices(e);
143 auto v1 = adjVertices.first;
144 auto v2 = adjVertices.second;
145 g(v1) = phi(v1);
146 g(v2) = phi(v2);
147 b(v1) = 1;
148 b(v2) = 1;
149 }
150 // Solve Î”u=0 with g as boundary conditions
151 EigenLinearAlgebraBackend::SolverSimplicialLDLT solver;
152 CFEM::LinearOperator L_dirichlet = DC::dirichletOperator( L, b );
153 solver.compute( L_dirichlet );
154 REQUIRE( solver.info() == Eigen::Success );
155 DenseVector g_dirichlet = DC::dirichletVector( L, g, b, g );
156 DenseVector x_dirichlet = solver.solve( g_dirichlet );
157 REQUIRE( solver.info() == Eigen::Success );
158 DenseVector u = DC::dirichletSolution( x_dirichlet, b, g );
159 double min_phi = 0.0;
160 double max_phi = 0.0;
161 double min_u = 0.0;
162 double max_u = 0.0;
163 double min_i_u = 0.0;
164 double max_i_u = 0.0;
165 for ( Index v = 0; v < surfmesh.nbVertices(); ++v )
166 {
167 min_phi = std::min( min_phi, phi( v ) );
168 max_phi = std::max( max_phi, phi( v ) );
169 min_u = std::min( min_u , u ( v ) );
170 max_u = std::max( max_u , u ( v ) );
171 if ( b( v ) == 0.0 )
172 {
173 min_i_u = std::min( min_i_u, u ( v ) );
174 max_i_u = std::max( max_i_u, u ( v ) );
175 }
176 }
177 REQUIRE( min_phi <= min_u );
178 REQUIRE( max_phi >= max_u );
179 REQUIRE( min_phi < min_i_u );
180 REQUIRE( max_phi > max_i_u );
181 } // for ( double scale = 0.1; scale < 2.0; scale *= 2.0 )
182 }
183};
