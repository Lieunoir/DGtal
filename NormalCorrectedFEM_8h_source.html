76 void buildTriangleFaceMass( Triplets & triplets, Triplets & lumpedTriplets,
77 Face f ) const
78 {
79 const auto vtcs = myMesh->incidentVertices( f );
80 const auto v0 = vtcs[ 0 ];
81 const auto v1 = vtcs[ 1 ];
82 const auto v2 = vtcs[ 2 ];
83 const auto xij = myMesh->position( v1 ) - myMesh->position( v0 );
84 const auto xik = myMesh->position( v2 ) - myMesh->position( v0 );
85 const auto n = xij.crossProduct( xik );
86 const auto nn = n.getNormalized();
87 const auto face_n = myMesh->faceNormal( f );
88 const double z = face_n.dot( nn );
89 const double doubleArea = n.norm();
90 triplets.push_back( { v0, v0, 1. / 12. * z * doubleArea } );
91 triplets.push_back( { v0, v1, 1. / 24. * z * doubleArea } );
92 triplets.push_back( { v0, v2, 1. / 24. * z * doubleArea } );
93 triplets.push_back( { v1, v0, 1. / 24. * z * doubleArea } );
94 triplets.push_back( { v1, v1, 1. / 12. * z * doubleArea } );
95 triplets.push_back( { v1, v2, 1. / 24. * z * doubleArea } );
96 triplets.push_back( { v2, v0, 1. / 24. * z * doubleArea } );
97 triplets.push_back( { v2, v1, 1. / 24. * z * doubleArea } );
98 triplets.push_back( { v2, v2, 1. / 12. * z * doubleArea } );
99 lumpedTriplets.push_back( { v0, v0, 1. / 6. * z * doubleArea } );
100 lumpedTriplets.push_back( { v1, v1, 1. / 6. * z * doubleArea } );
101 lumpedTriplets.push_back( { v2, v2, 1. / 6. * z * doubleArea } );
102 }

104 void buildQuadrangleFaceMass( Triplets & triplets,
105 Triplets & lumpedTriplets, Face f ) const
106 {
107 const auto vtcs = myMesh->incidentVertices( f );
108 const auto v0 = vtcs[ 0 ];
109 const auto v1 = vtcs[ 1 ];
110 const auto v2 = vtcs[ 2 ];
111 const auto v3 = vtcs[ 3 ];
112 const auto xij = myMesh->position( v1 ) - myMesh->position( v0 );
113 const auto xil = myMesh->position( v3 ) - myMesh->position( v0 );
114 const auto nn = xij.crossProduct( xil ).getNormalized();
115 const auto face_n = myMesh->faceNormal( f );
116 const double z = nn.dot( face_n );
117 if ( z < 0. )
118 {
119 trace.warning() << "Bad corrected normals at face " << f << std::endl;
120 }
121
122 double area = xij.crossProduct( xil ).norm();
123 int v[ 2 ][ 2 ] = { { (int)v0, (int)v1 }, { (int)v3, (int)v2 } };
124 for ( int i = 0; i < 2; i++ )
125 {
126 for ( int j = 0; j < 2; j++ )
127 {
128 for ( int i2 = 0; i2 < 2; i2++ )
129 {
130 for ( int j2 = 0; j2 < 2; j2++ )
131 {
132 int v_i = v[ i ][ j ];
133 int v_j = v[ i2 ][ j2 ];
134 if ( i == i2 && j == j2 )
135 {
136 triplets.push_back( { v_i, v_j, area * z * 1. / 9. } );
137 lumpedTriplets.push_back( { v_i, v_j, 1. / 4. * z * area } );
138 }
139 else if ( i == i2 && j == 1 - j2 )
140 {
141 triplets.push_back( { v_i, v_j, area * z * 1. / 18. } );
142 }
143 else if ( j == j2 && i == 1 - i2 )
144 {
145 triplets.push_back( { v_i, v_j, area * z * 1. / 18. } );
146 }
147 else
148 {
149 triplets.push_back( { v_i, v_j, area * z * 1. / 36. } );
150 }
151 }
152 }
153 }
154 }

157 DenseMatrix buildTriangleFaceStiffness( Face f ) const
158 {
159 DenseMatrix res = DenseMatrix( 3, 3 );
160 const auto vtcs = myMesh->incidentVertices( f );
161 const auto v0 = vtcs[ 0 ];
162 const auto v1 = vtcs[ 1 ];
163 const auto v2 = vtcs[ 2 ];
164 const auto xij = myMesh->position( v1 ) - myMesh->position( v0 );
165 const auto xik = myMesh->position( v2 ) - myMesh->position( v0 );
166 const auto n = xij.crossProduct( xik );
167 const auto nn = n.getNormalized();
168 const auto xil = nn.crossProduct( xij ).getNormalized();
169 const auto face_n = myMesh->faceNormal( f );
170 const double x = face_n.dot( xij.getNormalized() );
171 const double y = face_n.dot( xil );
172 const double z = face_n.dot( nn );
173 const double l1 = xij.norm();
174 const double l2 = xik.norm();
175 const double cos_t = xij.dot( xik ) / ( l1 * l2 );
176 const double sin_t = n.norm() / ( l1 * l2 );
177 const double x2 = x * x;
178 const double y2 = y * y;
179 const double z2 = z * z;
180 const double l12 = l1 * l1;
181 const double l22 = l2 * l2;
182 const double cos_t2 = cos_t * cos_t;
183 const double sin_t2 = sin_t * sin_t;
184 res( 0, 0 ) =
185 1. / 2 *
186 ( 2 * ( l1 * l2 - cos_t * l22 ) * sin_t * x * y - 2 * cos_t * l1 * l2 +
187 l22 * sin_t2 * z2 - l22 * sin_t2 +
188 ( 2 * cos_t * l1 * l2 + 2 * l22 * sin_t2 - l12 - l22 ) * x2 + l12 +
189 l22 ) /
190 ( l1 * l2 * sin_t * z );
191 res( 1, 0 ) =
192 1. / 2 *
193 ( ( 2 * cos_t * l2 - l1 ) * sin_t * x * y + l2 * sin_t2 * y2 +
194 cos_t * l1 - ( cos_t * l1 + l2 * sin_t2 - l2 ) * x2 - l2 ) /
195 ( l1 * sin_t * z );
196 res( 2, 0 ) =
197 -1. / 2 *
198 ( l2 * sin_t * x * y - cos_t * l2 + ( cos_t * l2 - l1 ) * x2 + l1 ) /
199 ( l2 * sin_t * z );
200 res( 0, 1 ) =
201 1. / 2 *
202 ( ( 2 * cos_t * l2 - l1 ) * sin_t * x * y + l2 * sin_t2 * y2 +
203 cos_t * l1 - ( cos_t * l1 + l2 * sin_t2 - l2 ) * x2 - l2 ) /
204 ( l1 * sin_t * z );
205 res( 1, 1 ) = -1. / 2 *
206 ( 2 * cos_t * l2 * sin_t * x * y + cos_t2 * l2 * x2 -
207 ( cos_t2 - 1 ) * l2 * y2 - l2 ) /
208 ( l1 * sin_t * z );
209 res( 2, 1 ) =
210 1. / 2 * ( sin_t * x * y + cos_t * x2 - cos_t ) / ( sin_t * z );
211 res( 0, 2 ) =
212 -1. / 2 *
213 ( l2 * sin_t * x * y - cos_t * l2 + ( cos_t * l2 - l1 ) * x2 + l1 ) /
214 ( l2 * sin_t * z );
215 res( 1, 2 ) =
216 1. / 2 * ( sin_t * x * y + cos_t * x2 - cos_t ) / ( sin_t * z );
217 res( 2, 2 ) = -1. / 2 * ( l1 * x2 - l1 ) / ( l2 * sin_t * z );
218 return res;
219 }

221 DenseMatrix buildQuadrangleFaceStiffness( Face f ) const
222 {
223 DenseMatrix res = DenseMatrix( 4, 4 );
224 const auto vtcs = myMesh->incidentVertices( f );
225 const auto v0 = vtcs[ 0 ];
226 const auto v1 = vtcs[ 1 ];
227 const auto v2 = vtcs[ 2 ];
228 const auto v3 = vtcs[ 3 ];
229 const auto xij = myMesh->position( v1 ) - myMesh->position( v0 );
230 const auto xil = myMesh->position( v3 ) - myMesh->position( v0 );
231 const auto nn = xij.crossProduct( xil ).getNormalized();
232 const auto face_n = myMesh->faceNormal( f );
233 const double x = face_n.dot( xij.getNormalized() );
234 const double y = face_n.dot( xil.getNormalized() );
235 const double z = face_n.dot( nn );
236 if ( z < 0. )
237 {
238 trace.warning() << "Bad corrected normals at face " << f << std::endl;
239 }
240 const double lap = 0.5 * ( x * y / z );
241 const double sqx = 1. - x * x;
242 const double sqy = 1. - y * y;
243 const double self = lap + 1. / 3. * ( sqx + sqy ) / z;
244 const double op1 = -1. / 3. * sqy / z + 1. / 6. * sqx / z;
245 const double op2 = -1. / 3. * sqx / z + 1. / 6. * sqy / z;
246 const double op = -lap - 1. / 6. * ( sqx + sqy ) / z;
247 res( 0, 0 ) = self;
248 res( 0, 1 ) = op1;
249 res( 0, 2 ) = op;
250 res( 0, 3 ) = op2;
251
252 res( 1, 0 ) = op1;
253 res( 1, 1 ) = self - 2. * lap;
254 res( 1, 2 ) = op2;
255 res( 1, 3 ) = op + 2. * lap;
256
257 res( 2, 0 ) = op;
258 res( 2, 1 ) = op2;
259 res( 2, 2 ) = self;
260 res( 2, 3 ) = op1;
261
262 res( 3, 0 ) = op2;
263 res( 3, 1 ) = op + 2. * lap;
264 res( 3, 2 ) = op1;
265 res( 3, 3 ) = self - 2. * lap;
266 return res;
267 }

292 LinearOperator buildL0() const
293 {
294 LinearOperator L( myMesh->nbVertices(), myMesh->nbVertices() );
295 Triplets triplets;
296 DenseMatrix values;
297 for ( Face f = 0; f < myMesh->nbFaces(); f++ )
298 {
299 const auto vtcs = myMesh->incidentVertices( f );
300 if ( vtcs.size() == 3 )
301 values = buildTriangleFaceStiffness( f );
302 else if ( vtcs.size() == 4 )
303 values = buildQuadrangleFaceStiffness( f );
304 else
305 {
306 trace.error() << "[NormalCorrectedFEM::init]"
307 << " faces should be triangle or quadrangles"
308 << std::endl;
309
310 return L;
311 }
312 for ( int i = 0; i < values.rows(); i++ )
313 for ( int j = 0; j < values.cols(); j++ )
314 triplets.push_back( { vtcs[ i ], vtcs[ j ], values( i, j ) } );
315 }
316 L.setFromTriplets( triplets.cbegin(), triplets.cend() );
317 return -L;
318 }
